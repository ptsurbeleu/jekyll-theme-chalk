export var polyfill = (function () {
	var clock = Date.now()
 
	return function (callback) {
		var currentTime = Date.now()
		if (currentTime - clock > 16) {
			clock = currentTime
			callback(currentTime)
		} else {
			setTimeout(function () { return polyfill(callback); }, 0)
		}
	}
})()
